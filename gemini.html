程式碼範例
使用 AI Studio API Key 直接呼叫
Python
語氣控制
多說話者
import urllib.request, json, base64, wave, sys
sys.stdout.reconfigure(encoding='utf-8')

API_KEY = "你的 Gemini API Key"
TEXT = "你好，這是 Gemini TTS 語音合成測試"

body = json.dumps({
    "contents": [{"parts": [{"text": TEXT}]}],
    "generationConfig": {
        "responseModalities": ["AUDIO"],
        "speechConfig": {
            "voiceConfig": {
                "prebuiltVoiceConfig": {"voiceName": "Leda"}  # 選擇聲音
            }
        }
    }
}, ensure_ascii=False).encode('utf-8')

req = urllib.request.Request(
    f"https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-tts:generateContent?key={API_KEY}",
    data=body,
    headers={"Content-Type": "application/json"}
)
r = urllib.request.urlopen(req)
d = json.loads(r.read())
pcm = base64.b64decode(d["candidates"][0]["content"]["parts"][0]["inlineData"]["data"])

# 儲存為 WAV
with wave.open("output.wav", "wb") as wf:
    wf.setnchannels(1)    # 單聲道
    wf.setsampwidth(2)    # 16-bit
    wf.setframerate(24000) # 24kHz
    wf.writeframes(pcm)

print("✅ 已儲存為 output.wav")
# Gemini TTS 支援自然語言控制語氣！
# 直接在文字前加上描述即可

examples = [
    # 興奮語氣
    "Say this with excitement: 我們贏了！太棒了！",

    # 溫柔語氣
    "Say this gently and softly: 晚安，好好休息喔",

    # 專業播報
    "Say this like a news anchor: 今日天氣晴朗，氣溫攝氏25度",

    # 說故事
    "Tell this like a bedtime story: 從前從前，有一隻小熊...",
]

# 使用方式和一般呼叫相同
body = json.dumps({
    "contents": [{"parts": [{"text": "Say this warmly and lovingly: 親愛的，今天辛苦了，早點休息吧"}]}],
    "generationConfig": {
        "responseModalities": ["AUDIO"],
        "speechConfig": {"voiceConfig": {"prebuiltVoiceConfig": {"voiceName": "Leda"}}}
    }
}, ensure_ascii=False).encode('utf-8')
# 多說話者對話合成
body = json.dumps({
    "contents": [{"parts": [{"text": "TTS the following conversation between Joe and Jane:\nJoe: How are you today?\nJane: I'm doing great, thank you!"}]}],
    "generationConfig": {
        "responseModalities": ["AUDIO"],
        "speechConfig": {
            "multiSpeakerVoiceConfig": {
                "speakerVoiceConfigs": [
                    {
                        "speaker": "Joe",
                        "voiceConfig": {"prebuiltVoiceConfig": {"voiceName": "Achird"}}
                    },
                    {
                        "speaker": "Jane",
                        "voiceConfig": {"prebuiltVoiceConfig": {"voiceName": "Leda"}}
                    }
                ]
            }
        }
    }
}, ensure_ascii=False).encode('utf-8')
